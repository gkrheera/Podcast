Netlify Function Setup

Create this file alongside your index.html to proxy Gemini through a server-side env variable:

netlify/functions/gemini.js
exports.handler = async (event) => { if (event.httpMethod === 'OPTIONS') { return { statusCode: 200, headers: { 'Access-Control-Allow-Origin': '*', 'Access-Control-Allow-Headers': 'Content-Type' }, body: '' }; } const key = process.env.GEMINI_API_KEY; if (!key) return { statusCode: 500, body: 'GEMINI_API_KEY not set' }; const url = 'https://generativelanguage.googleapis.com/v1beta' + '/models/gemini-2.0-flash:generateContent?key=' + key; const r = await fetch(url, { method: 'POST', headers: { 'Content-Type': 'application/json' }, body: event.body }); const d = await r.json(); return { statusCode: r.status, headers: { 'Content-Type': 'application/json', 'Access-Control-Allow-Origin': '*' }, body: JSON.stringify(d) }; };

Then: Netlify dashboard → Site settings → Environment variables → Add GEMINI_API_KEY
